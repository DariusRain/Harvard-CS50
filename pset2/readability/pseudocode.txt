	VOID ANALYZE-INPUT(STRING INPUT-STRING):

		FOR (INT INDEX = 0, INT STRING-LENGTH = STERLEN(INPUT-STRING); INDEX < STRING-LENGTH; INDEX + 1):

				IF "?" || "." || "!":

					++NUMBER-OF-SENTENCES;

				ELSE IF " ":

					IF INPUT[INDEX + 1] != " ";
						++NUMBER-OF-WORDS;

				ELSE IF IS-ALPHA(INPUT[INDEX]) && !IS-DIGIT(INPUT[INDEX])
					IF INDEX == 0
						++NUMBER-OF-WORDS

					++NUMBER-OF-LETTERS


	VOID LIAU-ALG(INT NUMBER-OF-LETTERS, INT NUMBER-OF-WORDS, INT NUMBER-OF-SENTENCES):

		FLOAT SENTENCES-PER-HUNDRED-WORDS = (NUMBER-OF-SENTENCES / NUMBER-OF-WORDS) * 100;

		FLOAT LETTERS-PER-HUNDRED-WORDS = (NUMBER-OF-LETTERS / NUMBER-OF-WORDS) * 100;

		FLOAT LIAU-PERSICE = 0.0588 * LETTERS-PER-HUNDRED-WORDS - 0.296 * SENTENCES-PER-HUNDRED-WORDS;

		INT LIAU-ROUNDED = ROUND(LIAU-PERSICE);

		PRINTF	Grade %i, LIAU-ROUNDED;
